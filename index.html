export function debounce<A extends any[], R extends any>(
  _target: Object,
  _propertyKey: string | symbol,
  descriptor: TypedPropertyDescriptor<(...args: A) => R>
) {
  const fn = descriptor.value!

  let isUsing = false

  return {
    ...descriptor,
    async value(...args: A): Promise<Awaited<R>> {
      if (isUsing) throw new Error("The function call is in progress")

      try {
        isUsing = true
        return await fn.call(this, ...args)
      } finally {
        isUsing = false
      }
    },
  }
}
            
          
            
import { CACHE_NAME } from "./constants"

// https://github.com/microsoft/TypeScript/issues/14877#issuecomment-1057795918
declare const self: ServiceWorkerGlobalScope

self.addEventListener("install", () => {
  self.skipWaiting()
  console.debug("service worker has been installed")
})

self.addEventListener("activate", (event) => {
  event.waitUntil(self.clients.claim())
  console.debug("service worker has been activated", CACHE_NAME)
})

self.addEventListener("fetch", (event) => {
  const { request } = event
  const { clientId } = event

  if (request.method !== "GET") return

    event.respondWith(
      caches.match(request, { cacheName: CACHE_NAME }).then(
        (response) =>
          response ||
          fetch(request).then(() =>
            caches.open(CACHE_NAME).then((cache) => {
              cache.put(request, response!.clone()).then(() =>
                self.clients.get(clientId).then((client) =>
                  client!.postMessage({
                    type: "cached",
                    payload: request.url,
                  })
                )
              )
              return response!
            })
          )
      )
    )
})
